  1. 从首页进入能够浏览到所有专题（已在上一章首页配置中实现了）

  2. 头部导航，也是导航到所有专题

    // 在 docs\.vuepress\config\navBar.js 维护了导航
/**
 * 导航栏维护
 */
const titles = require('../autoSiderbar/textTitle');
navbar = [
    {
        text: titles['topicNav'],
        link: '/topicNav/',
    }
]
module.exports = navbar;
